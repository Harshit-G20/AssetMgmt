const apiGatewayUrl = "https://xyz.execute-api.us-east-1.amazonaws.com/prod/upload-asset"; async function uploadFile() { const file = document.getElementById('fileUpload').files[0]; if (!file) { alert("Please select a file."); return; } const reader = new FileReader(); reader.readAsText(file); reader.onload = async function () { const fileContent = btoa(reader.result); // Convert to Base64 const assetId = file.name.split('.')[0]; // Extract asset ID from filename const response = await fetch(apiGatewayUrl, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ asset_id: assetId, file_content: fileContent }) }); if (response.ok) { alert("File uploaded successfully!"); } else { alert("Upload failed."); } }; }
